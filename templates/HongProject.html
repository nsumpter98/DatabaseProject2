Instructions for Handling Missing Data:¶

Substitution with Mean (Numeric Attributes):¶

  • Replace missing values in numeric attributes with the mean of non-missing values in the respective attribute.

Substitution with Median (Numeric Attributes):¶

  • Replace missing values in numeric attributes with the median of non-missing values in the respective attribute.

Substitution with Mode (Categorical Attributes):¶

  • Replace missing values in categorical attributes with the mode (most frequent value) of non-missing values in the respective attribute.

Predictive Modeling-Based Imputation:¶

Use a predictive model to estimate missing values in numeric attributes based on non-missing values and other attributes. You have the flexibility to design your predictive modeling-based imputation method, this can include techniques such as k-nearest neighbors, linear regression, or other algorithms. Example: Train a linear regression model using non-missing values to predict missing Age values.

"""
    Finds rows with missing values in a DataFrame and identifies the columns with missing data and their types.
    
    Parameters:
    - df (DataFrame): The DataFrame to analyze.
    
    Returns:
    - DataFrame with columns 'Index', 'Missing Columns', and 'Column Types'. Each row in the returned DataFrame
      corresponds to a row in the input DataFrame that contains missing values. 'Missing Columns' lists the names
      of columns with missing values. 'Column Types' lists the data types of these columns.
    """

def findMissingCategories(df):
    # Find rows with any missing values and columns with missing values for those rows
    missingCategoriesDf = df[df.isnull().any(axis=1)].isnull()
    missingCategoriesDf = missingCategoriesDf[missingCategoriesDf.any(axis=1)]

    # Collecting rows with missing information, including the data types of missing columns
    rowsToAppend = []
    for index, row in missingCategoriesDf.iterrows():
        missingColumns = ', '.join(row[row].index.tolist())
        # For each missing column, get the data type and concatenate it with the column name
        columnTypes = ', '.join([f"{str(df[col].dtype)}" for col in row[row].index])
        rowsToAppend.append({'Index': index, 'Missing Columns': missingColumns, 'Column Types': columnTypes})

    # Creating a DataFrame from the collected rows
    missingInfoDf = pd.DataFrame(rowsToAppend)

    return missingInfoDf

def missingNumericToMean(df, decimalPlaces = 0):
    """
    Creates a copy of the input DataFrame, replaces missing values in numeric columns with the mean
    of those columns, and returns the modified copy.
    
    Parameters:
    - df (DataFrame): The DataFrame in which to replace missing values.
    
    Returns:
    - DataFrame: A copy of the input DataFrame with missing values in numeric columns replaced by the mean.
    """
    dfCopy = df.copy()

    # Identify numeric columns (float64, int64) in the DataFrame copy
    numericColumns = dfCopy.select_dtypes(include=['float64', 'int64']).columns

    # Loop through the numeric columns and replace missing values with the mean of each column
    for column in numericColumns:
        if dfCopy[column].isnull().any():  # Check if the column has any missing values
            roundedMean = round(dfCopy[column].mean(), decimalPlaces)
            dfCopy[column].fillna(roundedMean, inplace=True)

    return dfCopy

def missingNumericToMedian(df, decimalPlaces = 0):
    """
    Creates a copy of the input DataFrame, replaces missing values in numeric columns with the median
    of those columns, and returns the modified copy.
    
    Parameters:
    - df (DataFrame): The DataFrame in which to replace missing values.
    
    Returns:
    - DataFrame: A copy of the input DataFrame with missing values in numeric columns replaced by the median.
    """
    dfCopy = df.copy()

    # Identify numeric columns (float64, int64) in the DataFrame copy
    numericColumns = dfCopy.select_dtypes(include=['float64', 'int64']).columns

    # Loop through the numeric columns and replace missing values with the mean of each column
    for column in numericColumns:
        if dfCopy[column].isnull().any():  # Check if the column has any missing values
            roundedMedian = round(dfCopy[column].median(), decimalPlaces)
            dfCopy[column].fillna(roundedMedian, inplace=True)
            
    return dfCopy

def missingCategoricalToMode(df):
    """
    Creates a copy of the input DataFrame, replaces missing values in object/string columns with the mode
    of those columns, and returns the modified copy.
    
    Parameters:
    - df (DataFrame): The DataFrame in which to replace missing values.
    
    Returns:
    - DataFrame: A copy of the input DataFrame with missing values in object/string columns replaced by the mode.
    """
    dfCopy = df.copy()

    # Identify numeric columns (float64, int64) in the DataFrame copy
    numericColumns = dfCopy.select_dtypes(include=['object']).columns
    # Loop through the numeric columns and replace missing values with the mean of each column
    for column in numericColumns:
        if dfCopy[column].isnull().any():  # Check if the column has any missing values
            mode = dfCopy[column].mode()[0]
            dfCopy[column].fillna(mode, inplace=True)

    return dfCopy

def imputeMissingCategoriesWithRF(df, columnToImpute):
    dfCopy = df.copy()

    # Encode all categorical columns
    labelEncoders = {}
    for col in dfCopy.select_dtypes(include=['object']).columns:
        le = LabelEncoder()
        # Directly encode the column, temporarily treating NaNs as a separate category
        # This ensures 'Missing' becomes part of the categories if NaNs exist
        dfCopy[col] = dfCopy[col].fillna('Missing')
        dfCopy[col] = le.fit_transform(dfCopy[col])
        labelEncoders[col] = le

    # If columnToImpute had 'Missing' encoded, find its index. Otherwise, set to None
    missingLabelIndex = labelEncoders[columnToImpute].transform(['Missing'])[0] if 'Missing' in labelEncoders[columnToImpute].classes_ else None

    # Prepare the data
    if missingLabelIndex is not None:
        missingIndexes = dfCopy[columnToImpute] == missingLabelIndex
        X_train = dfCopy.loc[~missingIndexes].drop(columns=[columnToImpute])
        y_train = dfCopy.loc[~missingIndexes, columnToImpute]
        X_missing = dfCopy.loc[missingIndexes].drop(columns=[columnToImpute])

        if not X_missing.empty:
            clf = RandomForestClassifier(n_estimators=100, random_state=42)
            clf.fit(X_train, y_train)

            # Predict missing values
            predicted_values = clf.predict(X_missing)
            dfCopy.loc[missingIndexes, columnToImpute] = predicted_values

    # Inverse transform the encoded columns back to their original categories
    for col, le in labelEncoders.items():
        dfCopy[col] = le.inverse_transform(dfCopy[col])
        # Replace 'Missing' placeholder back to np.nan if 'Missing' was used
        if 'Missing' in le.classes_:
            dfCopy[col] = dfCopy[col].replace('Missing', np.nan)

    return dfCopy

# Applying the function to all object type columns in dfFinalStructured
for column in dfFinalStructured.select_dtypes(include=['object']).columns:

    dfFinalStructured = imputeMissingCategoriesWithRF(dfFinalStructured, column)
